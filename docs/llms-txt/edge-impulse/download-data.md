# Download data

> Download the processed data for this learning block. This is data already processed by the signal processing blocks.



## OpenAPI

````yaml .assets/openapi.yaml get /api/{projectId}/training/{learnId}/x
openapi: 3.0.0

  title: Edge Impulse API
  version: 1.0.0
servers:
  - url: https://studio.edgeimpulse.com/v1

  - JWTAuthentication: []
  - JWTHttpHeaderAuthentication: []
  - OAuth2: []
paths:
  /api/{projectId}/training/{learnId}/x:
    get:
      tags:
        - Learn
      summary: Download data
      description: >-
        Download the processed data for this learning block. This is data

      operationId: getLearnXData
      parameters:
        - $ref: '#/components/parameters/ProjectIdParameter'
        - $ref: '#/components/parameters/LearnIdParameter'
      responses:
        '200':
          description: Numpy binary file
          content:
            application/octet-stream:
              schema:
                type: string
                format: binary
components:
  parameters:

      name: projectId
      in: path
      required: true

      schema:

      in: path
      required: true

        type: integer

      type: apiKey

          tokenUrl: /v1/oauth/token
          scopes:
````
