type ConnectionStatusAttr [string](https://pkg.go.dev/builtin#string)

type ServerActiveConnections struct {
[metric](https://pkg.go.dev/go.opentelemetry.io/otel/metric).[Int64UpDownCounter](https://pkg.go.dev/go.opentelemetry.io/otel/metric#Int64UpDownCounter)}
