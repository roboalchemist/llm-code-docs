#### type [FieldOrder](https://github.com/temporalio/api-go/blob/v1.62.2/internal/protojson/order/order.go#L13)[¶](https://pkg.go.dev/go.temporal.io/api@v1.62.2/internal/protojson/order#FieldOrder "Go to FieldOrder")

type FieldOrder func(x, y [protoreflect](https://pkg.go.dev/google.golang.org/protobuf/reflect/protoreflect).[FieldDescriptor](https://pkg.go.dev/google.golang.org/protobuf/reflect/protoreflect#FieldDescriptor)) [bool](https://pkg.go.dev/builtin#bool)

FieldOrder specifies the ordering to visit message fields. It is a function that reports whether x is ordered before y.

var (
// AnyFieldOrder specifies no specific field ordering.	AnyFieldOrder [FieldOrder](https://pkg.go.dev/go.temporal.io/api@v1.62.2/internal/protojson/order#FieldOrder) = [nil](https://pkg.go.dev/builtin#nil)

// LegacyFieldOrder sorts fields in the same ordering as emitted by	// wire serialization in the github.com/golang/protobuf implementation.
	LegacyFieldOrder [FieldOrder](https://pkg.go.dev/go.temporal.io/api@v1.62.2/internal/protojson/order#FieldOrder) = func(x, y [protoreflect](https://pkg.go.dev/google.golang.org/protobuf/reflect/protoreflect).[FieldDescriptor](https://pkg.go.dev/google.golang.org/protobuf/reflect/protoreflect#FieldDescriptor)) [bool](https://pkg.go.dev/builtin#bool) {
		ox, oy := x.ContainingOneof(), y.ContainingOneof()
		inOneof := func(od [protoreflect](https://pkg.go.dev/google.golang.org/protobuf/reflect/protoreflect).[OneofDescriptor](https://pkg.go.dev/google.golang.org/protobuf/reflect/protoreflect#OneofDescriptor)) [bool](https://pkg.go.dev/builtin#bool) {
			return od != [nil](https://pkg.go.dev/builtin#nil) && !od.IsSynthetic()
		}

		if x.IsExtension() != y.IsExtension() {
			return x.IsExtension() && !y.IsExtension()
		}

		if inOneof(ox) != inOneof(oy) {
			return !inOneof(ox) && inOneof(oy)
		}

		if inOneof(ox) && inOneof(oy) && ox != oy {
			return ox.Index() < oy.Index()
		}

		return x.Number() < y.Number()
	}

// NumberFieldOrder sorts fields by their field number.	NumberFieldOrder [FieldOrder](https://pkg.go.dev/go.temporal.io/api@v1.62.2/internal/protojson/order#FieldOrder) = func(x, y [protoreflect](https://pkg.go.dev/google.golang.org/protobuf/reflect/protoreflect).[FieldDescriptor](https://pkg.go.dev/google.golang.org/protobuf/reflect/protoreflect#FieldDescriptor)) [bool](https://pkg.go.dev/builtin#bool) {
		return x.Number() < y.Number()
	}

// IndexNameFieldOrder sorts non-extension fields before extension fields.	// Non-extensions are sorted according to their declaration index.
	// Extensions are sorted according to their full name.
	IndexNameFieldOrder [FieldOrder](https://pkg.go.dev/go.temporal.io/api@v1.62.2/internal/protojson/order#FieldOrder) = func(x, y [protoreflect](https://pkg.go.dev/google.golang.org/protobuf/reflect/protoreflect).[FieldDescriptor](https://pkg.go.dev/google.golang.org/protobuf/reflect/protoreflect#FieldDescriptor)) [bool](https://pkg.go.dev/builtin#bool) {

		if x.IsExtension() != y.IsExtension() {
			return !x.IsExtension() && y.IsExtension()
		}

		if x.IsExtension() && y.IsExtension() {
			return x.FullName() < y.FullName()
		}

		return x.Index() < y.Index()
	}
)

#### type [KeyOrder](https://github.com/temporalio/api-go/blob/v1.62.2/internal/protojson/order/order.go#L67)[¶](https://pkg.go.dev/go.temporal.io/api@v1.62.2/internal/protojson/order#KeyOrder "Go to KeyOrder")

type KeyOrder func(x, y [protoreflect](https://pkg.go.dev/google.golang.org/protobuf/reflect/protoreflect).[MapKey](https://pkg.go.dev/google.golang.org/protobuf/reflect/protoreflect#MapKey)) [bool](https://pkg.go.dev/builtin#bool)

KeyOrder specifies the ordering to visit map entries. It is a function that reports whether x is ordered before y.

var (
// AnyKeyOrder specifies no specific key ordering.	AnyKeyOrder [KeyOrder](https://pkg.go.dev/go.temporal.io/api@v1.62.2/internal/protojson/order#KeyOrder) = [nil](https://pkg.go.dev/builtin#nil)

// GenericKeyOrder sorts false before true, numeric keys in ascending order,	// and strings in lexicographical ordering according to UTF-8 codepoints.
	GenericKeyOrder [KeyOrder](https://pkg.go.dev/go.temporal.io/api@v1.62.2/internal/protojson/order#KeyOrder) = func(x, y [protoreflect](https://pkg.go.dev/google.golang.org/protobuf/reflect/protoreflect).[MapKey](https://pkg.go.dev/google.golang.org/protobuf/reflect/protoreflect#MapKey)) [bool](https://pkg.go.dev/builtin#bool) {
		switch x.Interface().(type) {
		case [bool](https://pkg.go.dev/builtin#bool):
			return !x.Bool() && y.Bool()
		case [int32](https://pkg.go.dev/builtin#int32), [int64](https://pkg.go.dev/builtin#int64):
			return x.Int() < y.Int()
		case [uint32](https://pkg.go.dev/builtin#uint32), [uint64](https://pkg.go.dev/builtin#uint64):
			return x.Uint() < y.Uint()
		case [string](https://pkg.go.dev/builtin#string):
			return x.String() < y.String()
		default:
			[panic](https://pkg.go.dev/builtin#panic)("invalid map key type")
		}
	}
)
