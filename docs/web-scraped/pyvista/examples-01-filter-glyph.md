Use vectors in a dataset to plot and orient glyphs/geometric objects.

from  __future__  import annotations

import math

import numpy as np

import pyvista as pv
from pyvista import examples

Glyphying can be done via the [`pyvista.DataSetFilters.glyph()`](https://docs.pyvista.org/api/core/_autosummary/pyvista.datasetfilters.glyph#pyvista.DataSetFilters.glyph "pyvista.DataSetFilters.glyph") filter

[mesh](https://docs.pyvista.org/api/core/_autosummary/pyvista.polydata#pyvista.PolyData "pyvista.PolyData") = examples.download_carotid().threshold(145, scalars='scalars')
[mask](https://docs.pyvista.org/api/utilities/_autosummary/pyvista.pyvista_ndarray#pyvista.pyvista_ndarray "pyvista.pyvista_ndarray") = [mesh](https://docs.pyvista.org/api/core/_autosummary/pyvista.polydata#pyvista.PolyData "pyvista.PolyData")['scalars'] < 210
[mesh](https://docs.pyvista.org/api/core/_autosummary/pyvista.polydata#pyvista.PolyData "pyvista.PolyData")['scalars'][[mask](https://docs.pyvista.org/api/utilities/_autosummary/pyvista.pyvista_ndarray#pyvista.pyvista_ndarray "pyvista.pyvista_ndarray")] = 0  # null out smaller vectors

# Make a geometric object to use as the glyph
[geom](https://docs.pyvista.org/api/core/_autosummary/pyvista.polydata#pyvista.PolyData "pyvista.PolyData") = [pv.Arrow](https://docs.pyvista.org/api/utilities/_autosummary/pyvista.arrow#pyvista.Arrow "pyvista.Arrow")()  # This could be any dataset

# Perform the glyph
[glyphs](https://docs.pyvista.org/api/core/_autosummary/pyvista.polydata#pyvista.PolyData "pyvista.PolyData") = [mesh.glyph](https://docs.pyvista.org/api/core/_autosummary/pyvista.datasetfilters.glyph#pyvista.DataSetFilters.glyph "pyvista.DataSetFilters.glyph")(orient='vectors', scale='scalars', factor=0.003, [geom](https://docs.pyvista.org/api/core/_autosummary/pyvista.polydata#pyvista.PolyData "pyvista.PolyData")=[geom](https://docs.pyvista.org/api/core/_autosummary/pyvista.polydata#pyvista.PolyData "pyvista.PolyData"))

# plot using the plotting class
[pl](https://docs.pyvista.org/api/plotting/_autosummary/pyvista.plotter#pyvista.Plotter "pyvista.Plotter") = [pv.Plotter](https://docs.pyvista.org/api/plotting/_autosummary/pyvista.plotter#pyvista.Plotter "pyvista.Plotter")()
[pl.add_mesh](https://docs.pyvista.org/api/plotting/_autosummary/pyvista.plotter.add_mesh#pyvista.Plotter.add_mesh "pyvista.Plotter.add_mesh")([glyphs](https://docs.pyvista.org/api/core/_autosummary/pyvista.polydata#pyvista.PolyData "pyvista.PolyData"), show_scalar_bar=False, lighting=False, cmap='coolwarm')
[pl.camera_position](https://docs.pyvista.org/api/plotting/_autosummary/pyvista.plotter.camera_position#pyvista.Plotter.camera_position "pyvista.Plotter.camera_position") = [pv.CameraPosition](https://docs.pyvista.org/api/plotting/_autosummary/pyvista.cameraposition#pyvista.CameraPosition "pyvista.CameraPosition")(
    position=(146.53, 91.28, 21.70),
    focal_point=(125.00, 94.45, 19.81),
    viewup=(-0.086, 0.007, 0.996),
)  # view only part of the vector field
[cpos](https://docs.pyvista.org/api/plotting/_autosummary/pyvista.cameraposition#pyvista.CameraPosition "pyvista.CameraPosition") = [pl.show](https://docs.pyvista.org/api/plotting/_autosummary/pyvista.plotter.show#pyvista.Plotter.show "pyvista.Plotter.show")(return_cpos=True)

Another approach is to load the vectors directly to the mesh object and then access the [`pyvista.DataSet.arrows`](https://docs.pyvista.org/api/core/_autosummary/pyvista.dataset.arrows#pyvista.DataSet.arrows "pyvista.DataSet.arrows") property.

[sphere](https://docs.pyvista.org/api/core/_autosummary/pyvista.polydata#pyvista.PolyData "pyvista.PolyData") = [pv.Sphere](https://docs.pyvista.org/api/utilities/_autosummary/pyvista.sphere#pyvista.Sphere "pyvista.Sphere")(radius=[math.pi](https://docs.python.org/3.11/library/functions.html#float "builtins.float"))

# make cool swirly pattern
[vectors](https://numpy.org/doc/stable/reference/generated/numpy.ndarray.html#numpy.ndarray "numpy.ndarray") = [np.vstack](https://numpy.org/doc/stable/reference/generated/numpy.vstack.html#numpy.vstack "numpy.vstack")(
    (
        [np.sin](https://numpy.org/doc/stable/reference/generated/numpy.ufunc.html#numpy.ufunc "numpy.ufunc")([sphere.points](https://docs.pyvista.org/api/core/_autosummary/pyvista.polydatafilters.points#pyvista.PolyDataFilters.points "pyvista.PolyDataFilters.points")[:, 0]),
        [np.cos](https://numpy.org/doc/stable/reference/generated/numpy.ufunc.html#numpy.ufunc "numpy.ufunc")([sphere.points](https://docs.pyvista.org/api/core/_autosummary/pyvista.polydatafilters.points#pyvista.PolyDataFilters.points "pyvista.PolyDataFilters.points")[:, 1]),
        [np.cos](https://numpy.org/doc/stable/reference/generated/numpy.ufunc.html#numpy.ufunc "numpy.ufunc")([sphere.points](https://docs.pyvista.org/api/core/_autosummary/pyvista.polydatafilters.points#pyvista.PolyDataFilters.points "pyvista.PolyDataFilters.points")[:, 2]),
    ),
).T

# add and scale
[sphere](https://docs.pyvista.org/api/core/_autosummary/pyvista.polydata#pyvista.PolyData "pyvista.PolyData")['vectors'] = [vectors](https://numpy.org/doc/stable/reference/generated/numpy.ndarray.html#numpy.ndarray "numpy.ndarray") * 0.3

# plot just the arrows

Sometimes you might not want glyphs for every node in the input dataset. In this case, you can choose to build glyphs for a subset of the input dataset by using a merging tolerance. Here we specify a merging tolerance of five percent which equates to five percent of the bounding box’s length.

# Example dataset with normals
[mesh](https://docs.pyvista.org/api/core/_autosummary/pyvista.polydata#pyvista.PolyData "pyvista.PolyData") = examples.load_random_hills()

# create a subset of arrows using the glyph filter
[arrows](https://docs.pyvista.org/api/core/_autosummary/pyvista.polydata#pyvista.PolyData "pyvista.PolyData") = [mesh.glyph](https://docs.pyvista.org/api/core/_autosummary/pyvista.datasetfilters.glyph#pyvista.DataSetFilters.glyph "pyvista.DataSetFilters.glyph")(scale='Normals', orient='Normals', tolerance=0.05)

[pl](https://docs.pyvista.org/api/plotting/_autosummary/pyvista.plotter#pyvista.Plotter "pyvista.Plotter") = [pv.Plotter](https://docs.pyvista.org/api/plotting/_autosummary/pyvista.plotter#pyvista.Plotter "pyvista.Plotter")()
