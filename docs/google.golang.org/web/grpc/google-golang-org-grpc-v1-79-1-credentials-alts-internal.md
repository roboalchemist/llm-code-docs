	ServerHandshake(ctx [context](https://pkg.go.dev/context).[Context](https://pkg.go.dev/context#Context)) ([net](https://pkg.go.dev/net).[Conn](https://pkg.go.dev/net#Conn), [credentials](https://pkg.go.dev/google.golang.org/grpc@v1.79.1/credentials).[AuthInfo](https://pkg.go.dev/google.golang.org/grpc@v1.79.1/credentials#AuthInfo), [error](https://pkg.go.dev/builtin#error))
// Close terminates the Handshaker. It should be called when the caller	// obtains the secure connection.
	Close()
}
