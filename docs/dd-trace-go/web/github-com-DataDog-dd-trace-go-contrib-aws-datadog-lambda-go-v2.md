// DatadogTraceEnabledEnvVar is the environment variable that enables Datadog tracing.	DatadogTraceEnabledEnvVar = "DD_TRACE_ENABLED"
// MergeXrayTracesEnvVar is the environment variable that enables the merging of X-Ray and Datadog traces.	MergeXrayTracesEnvVar = "DD_MERGE_XRAY_TRACES"
// UniversalInstrumentation is the environment variable that enables universal instrumentation with the DD Extension	UniversalInstrumentation = "DD_UNIVERSAL_INSTRUMENTATION"
// Initialize otel tracer provider if enabled	OtelTracerEnabled = "DD_TRACE_OTEL_ENABLED"
// FIPSModeEnvVar is the environment variable that determines whether to enable FIPS mode.	// Defaults to true in GovCloud regions and false otherwise.
	FIPSModeEnvVar = "DD_LAMBDA_FIPS_MODE"
